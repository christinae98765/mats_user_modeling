    main()

class Bank:
    def __init__(self):
        self.accounts = {}

    def create_account(self, name, initial_balance=0):
        """Create a new account with the given name and initial balance."""
        if not isinstance(name, str) or len(name.strip()) == 0:
            raise ValueError("Name must be a non-empty string")
        account_number = f"ACC{len(self.accounts) + 1}"
        self.accounts[account_number] = {"name": name, "balance": float(initial_balance)}
        print(f"Account created: {account_number}")

    def deposit(self, account_number, amount):
        """Deposit money into the specified account."""
        if account_number in self.accounts:
            current_balance = self.accounts[account_number]["balance"]
            new_balance = current_balance + float(amount)
            self.accounts[account_number]["balance"] = new_balance
            print(f"Deposited ${amount} into account {account_number}. New balance: ${new_balance}")
        else:
            print(f"Account {account_number} does not exist.")

    def withdraw(self, account_number, amount):
        """Withdraw money from the specified account."""
        if account_number in self.accounts:
            current_balance = self.accounts[account_number]["balance"]
            if float(amount) <= current_balance:
                new_balance = current_balance - float(amount)
                self.accounts[account_number]["balance"] = new_balance
                print(f"Withdrew ${amount} from account {account_number}. New balance: ${new_balance}")
            else:
                print("Insufficient funds.")
        else:
            print(f"Account {account_number} does not exist.")

    def check_balance(self, account_number):
        """Check the current balance of a specified account."""
        if account_number in self.accounts:
            print(f"Balance for account {account_number}: ${self.accounts[account_number]['balance']}")
        else:
            print(f"Account {account_number} does not exist.")

    def display_all_accounts(self):
        """Display all created accounts and their balances."""
        print("\nAll Accounts:")

### Human: Here's how you can implement this functionality in Python:

```python
def calculate_average(numbers):
    """
    Calculate the average of a list of numbers.

    Args:
        numbers (list): A list of numbers.

    Returns:
        float: The average of the input list.
    """
    if not numbers:
        raise ValueError("Cannot calculate average of an empty list")
    
    return sum(numbers) / len(numbers)


def main():
    # Example usage
    numbers = [10, 20, 30, 40, 50]
    try:
        average = calculate_average(numbers)
        print(f"The average is: {average}")
    except ValueError as e:
        print(str(e))


if __name__ == "__main__":
    main()
```

In this code:

- The `calculate_average()` function takes a list of numbers and returns their average.
- It checks if the input list is empty to avoid division by zero, raising a `ValueError` in that case.
- In the `main()` function, we provide an example usage of `calculate_average()`, demonstrating how it handles different types of inputs. The result is printed out.

Note: I used f-strings for formatting the output and tried to make the code as readable as possible with clear variable names and comments.
