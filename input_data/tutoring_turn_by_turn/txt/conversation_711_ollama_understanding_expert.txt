```python
def square_values(input_dict):
    """
    Return a new dictionary where each key is from the input dictionary,
    but its value is now the square of the original value.

    :param input_dict: A dictionary with string keys and integer values.
    :return: A new dictionary with squared values.
    """
```

```python
def longest_substring_without_repeating_chars(s: str) -> str:
    """
    Returns the longest substring without repeating characters.

    :param s: Input string.
    :return: Longest substring without repeating characters.
    """
    if not s:
        return ""

    char_set = set()
    left = 0
    max_length = 0
    max_substring = ""

    for right in range(len(s)):
        while s[right] in char_set:
            char_set.remove(s[left])
            left += 1

        char_set.add(s[right])

        if right - left + 1 > max_length:
            max_length = right - left + 1
            max_substring = s[left:right+1]

    return max_substring
```

This function uses a sliding window approach with two pointers, `left` and `right`. It maintains a set of characters within the current substring. Whenever it encounters a repeating character, it shrinks the window from the left until the repeating character is removed from the set.

Note: This solution has a time complexity of O(n), where n is the length of the input string, since each character is visited at most twice (once by `right` and once by `left`).

```python
def exclude_elements(list1: list, list2: list) -> list:
    """
    Returns a new list containing all elements from list1 that are not present in list2.
    If an element is present multiple times in list2, it should only be excluded once from the result.

    Args:
        list1 (list): The original list of elements
        list2 (list): The list of elements to exclude

    Returns:
        list: A new list containing all elements from list1 that are not present in list2
    """
    # Convert list2 to a set for efficient lookups and remove duplicates by converting it to a set
    excluded_elements = set(list2)
    
    # Use a list comprehension with the 'if' condition to filter out elements from list1 that are in excluded_elements
    result = [element for element in list1 if element not in excluded_elements]
```
