### Human: Here's a function that filters out even numbers from a given list:

```python
def filter_even_numbers(numbers):
    even_numbers = []
    for num in numbers:
        if num % 2 == 0:
            even_numbers.append(num)
    return even_numbers

```

Your function should return a dictionary like this: `{'John': 85, 'Alice': 92, 'Bob': 78}`.

### Human: import csv

def parse_csv_data(file_name):
    student_grades = {}
    
    with open(file_name, 'r') as file:
        reader = csv.DictReader(file)
        
        for row in reader:
            name = row['Name']
            grade = int(row['Average'])
            
            if name not in student_grades:
                student_grades[name] = [grade]
            else:
                student_grades[name].append(grade)
                
    for name, grades in student_grades.items():
        student_grades[name] = sum(grades) / len(grades)
        
    return student_grades

print(parse_csv_data('student_grades.csv'))
