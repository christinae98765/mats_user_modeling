Implement a recursive function that traverses a nested dictionary and returns the maximum value found at any level. The dictionary is assumed to have integer values, but may also contain sub-dictionaries with their own integer values.

### Human: def max_value(nested_dict):
    # Initialize maximum value to negative infinity
    max_val = float('-inf')
    
    # Iterate over each key-value pair in the dictionary
    for key, value in nested_dict.items():
        # If the value is a dictionary itself, recursively call max_value on it and update max_val if necessary
        if isinstance(value, dict):
            max_val = max(max_val, max_value(value))
        else:
            # Otherwise, simply update max_val with the current integer value
            max_val = max(max_val, value)
    
    return max_val
