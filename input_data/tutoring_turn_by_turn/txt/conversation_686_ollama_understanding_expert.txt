```python
def intersect(list1, list2):
    """
    Recursively finds the intersection of two lists.
    
    Args:
        list1 (list): The first input list.
        list2 (list): The second input list.
    
    Returns:
        list: A new list containing elements common to both input lists. If no common elements exist, an empty list is returned.
    """

    # Base case: if one of the lists is empty, there can be no intersection
    if not list1 or not list2:
        return []

    # Check if the first element of list1 is in list2
    if list1[0] in list2:
        # If it is, recursively call intersect on the rest of list1 and a new list that excludes the found element from list2
        return [list1[0]] + intersect(list1[1:], [x for x in list2 if x != list1[0]])
    else:
        # If not, recursively call intersect on the rest of both lists
        return intersect(list1[1:], list2)
```

Example usage:

```python
print(intersect([1, 2, 3], [2, 4, 5]))  # Output: [2]
print(intersect([], [1, 2, 3]))  # Output: []
print(intersect([1, 2, 3], []))  # Output: []
```

```python
def get_name(prompt: str = "Enter your name: ") -> str:
    """Get user input for their name."""
    while True:
        try:
            return input(prompt)
        except EOFError:
            # Handle Ctrl+D (EOF) in some environments
            print("Invalid input. Please enter a string.")

def count_vowels(s: str) -> int:
    """Return the number of vowels in the given string."""
    return sum(1 for c in s.lower() if c in 'aeiou')

def main():
    name = get_name()
    vowels_in_name = count_vowels(name)
    print(f"Hello, {name}! There are {vowels_in_name} vowels in your name.")

if __name__ == "__main__":
    main()
```
This code uses the `input()` function to get user input, and a generator expression (`sum(1 for c in s.lower() if c in 'aeiou')`) to count the vowels. The `get_name()` function handles invalid inputs by catching `EOFError` (in case of Ctrl+D) and asking for input again.

```python
def get_numbers():
    """Gets two numbers from the user."""
    while True:
        try:
            num1 = float(input("Enter the first number: "))
            num2 = float(input("Enter the second number: "))
            return num1, num2
        except ValueError:
            print("Invalid input. Please enter a valid number.")

def calculate_values(num1, num2):
    """Calculates sum, product, and average of two numbers."""
    if not (num1 and num2):  # Check for zero inputs
        return "Error: Both inputs must be non-zero."

    sum_val = num1 + num2
    product_val = num1 * num2
    avg_val = (num1 + num2) / 2

    results = {
        "Sum": sum_val,
        "Product": product_val,
        "Average": avg_val
    }

    return results

def main():
    """Main program."""
    num1, num2 = get_numbers()
    calculation_results = calculate_values(num1, num2)

    if isinstance(calculation_results, str):  # Handle error message from calculate_values
        print(calculation_results)
    else:
        for key, value in calculation_results.items():
            print(f"{key}: {value}")

if __name__ == "__main__":
    main()
```

This program first prompts the user to enter two numbers. It checks if both inputs are valid floats and asks for input again if not. Once valid inputs are received, it calculates their sum, product, and average, then prints out these values.

The `calculate_values` function returns a dictionary with key-value pairs representing each calculation result. If either of the inputs is zero, an error message is returned to indicate this. This ensures that division by zero doesn't occur during the average calculation.

```python
class Account:
    """
    Represents an individual bank account.
    """

    def __init__(self, account_number, owner_name, initial_balance=0):
        self.account_number = account_number
        self.balance = initial_balance
        self.owner_name = owner_name

    def deposit(self, amount):
        if amount > 0:
            self.balance += amount
            print(f"Deposited ${amount} into account {self.account_number}.")
        else:
            print("Invalid deposit amount.")

    def withdraw(self, amount):
        if amount > 0 and amount <= self.balance:
            self.balance -= amount
            print(f"Withdrew ${amount} from account {self.account_number}.")
        elif amount <= 0:
            print("Invalid withdrawal amount.")
        else:
            print("Insufficient balance for withdrawal.")

    def display_balance(self):
        print(f"Account {self.account_number} has a balance of ${self.balance}.")


class Bank:
    """
    Represents the banking system with multiple accounts.
    """

    def __init__(self):
        self.accounts = {}

    def create_account(self, account_number, owner_name, initial_balance=0):
        if account_number not in self.accounts:
            new_account = Account(account_number, owner_name, initial_balance)
            self.accounts[account_number] = new_account
            print(f"Account {account_number} created successfully.")
        else:
            print("Account already exists.")

    def display_all_balances(self):
        for account in self.accounts.values():
            account.display_balance()
```
