def square_integers(nested_list):
    new_list = []
    for element in nested_list:
        if isinstance(element, int):
            new_list.append(element ** 2)
        elif isinstance(element, list):

### Human: def longest_substring(s):
    max_length = 0
    max_substring = ""
    start_index = 0
    
    # Initialize a set to store unique characters in the current substring
    char_set = set()
    
    for end_index, char in enumerate(s):
        while char in char_set:
            # If we encounter a repeated character, remove characters from the start of the substring
            char_set.remove(s[start_index])
            start_index += 1
        
        # Add the current character to the set and update max_length and max_substring if necessary
        char_set.add(char)
        
        if end_index - start_index + 1 > max_length:
            max_length = end_index - start_index + 1
            max_substring = s[start_index:end_index+1]
    
    return max_substring
