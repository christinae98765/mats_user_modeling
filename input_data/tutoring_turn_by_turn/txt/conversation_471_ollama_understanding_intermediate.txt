### Human: def longest_substring(s):

    max_length = 0
    max_substring = ""
    char_set = set()

    for end in range(len(s)):
        while s[end] in char_set:
            char_set.remove(s[start])
            start += 1
        char_set.add(s[end])

    result = ""
    
    for i in range(len(input_str)):
