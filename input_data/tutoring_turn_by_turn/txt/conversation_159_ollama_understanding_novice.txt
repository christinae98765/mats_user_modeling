```python
def above_average_grades(grades):
    # calculate average score
    avg_score = sum(grades.values()) / len(grades)
    
    # create a new dictionary with students who scored above average?
    above_avg = {}
    for student, score in grades.items():
        if score > avg_score:
            above_avg[student] = score
    
    # sort the new dictionary by scores in descending order
    sorted_above_avg = sorted(above_avg.items(), key=lambda x: x[1], reverse=True)
    
    return dict(sorted_above_avg)

print(above_average_grades(student_grades))
```
